-- TEAM Table
CREATE TABLE team (
    team_id INT PRIMARY KEY,
    name VARCHAR(100) NOT NULL,
    coach_id INT UNIQUE,
    stadium VARCHAR(100),
    no_of_championships INT DEFAULT 0,
    FOREIGN KEY(coach_id) references coach(coach_id) ON DELETE SET NULL
);

-- COACH Table
CREATE TABLE coach (
    coach_id INT PRIMARY KEY,
    name VARCHAR(100) NOT NULL,
    tactical_style VARCHAR(100),
    total_wins INT DEFAULT 0,
    championships INT DEFAULT 0
);

-- PLAYER Table
CREATE TABLE player (
    player_id INT PRIMARY KEY,
    name VARCHAR(100) NOT NULL,
    active BOOLEAN DEFAULT TRUE,
    age INT CHECK (age >= 15),
    position VARCHAR(50),
    team_id INT,
    nationality VARCHAR(50),
    FOREIGN KEY (team_id) REFERENCES team(team_id) ON DELETE SET NULL,
);

-- STANDINGS Table
CREATE TABLE standings (
    team_id INT,
    season_year YEAR,
    rank INT,
    wins INT DEFAULT 0,
    loses INT DEFAULT 0,
    draws INT DEFAULT 0,
    past_5 VARCHAR(5), 
    PRIMARY KEY (team_id, season_year),
    FOREIGN KEY (team_id) REFERENCES team(team_id) ON DELETE CASCADE
);

-- STATS Table
CREATE TABLE stats (
    player_id INT,
    season_year YEAR,
    goals INT DEFAULT 0,
    assists INT DEFAULT 0,
    appearances INT DEFAULT 0,
    tackles INT DEFAULT 0,
    red_cards INT DEFAULT 0,
    yellow_cards INT DEFAULT 0,
    PRIMARY KEY (player_id, season_year),
    FOREIGN KEY (player_id) REFERENCES player(player_id) ON DELETE CASCADE
);

-- MATCHES Table
CREATE TABLE matches (
    match_id INT AUTO_INCREMENT PRIMARY KEY,
    home_team_id INT,
    away_team_id INT,
    match_date DATE,
    h_goals INT DEFAULT 0,
    a_goals INT DEFAULT 0,
    FOREIGN KEY (home_team_id) REFERENCES team(team_id) ON DELETE CASCADE,
    FOREIGN KEY (away_team_id) REFERENCES team(team_id) ON DELETE CASCADE,
    CHECK (home_team_id <> away_team_id)
);
